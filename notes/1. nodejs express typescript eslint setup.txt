Creating Backend
	create backend directory
	npm init -y [ creates package.json ]
	npm install --save-dev typescript [ Installs typescripts in node modules ]
	npx tsc --init [ creates tsconfig.json file, tsc stands for typescript compiler ]
	npm install express [ installs express.js ]
	npm i --save-dev @types/express

Run backend files
	node server.ts [ Works with javascript files only - need to convert .ts to .js]
	npx tsc  [ converts typescript code to javascript code ]
	node server.js

create directory src for ".ts" files
edit "outDir" as "./dist" so ".js" files generated by tsc will be collected there

Automate compilation of .ts files and run them using node
	npm install --save-dev nodemon [ automates running server using node - works with javascript ]
	npm install -D ts-node [converts .ts to .js and runs with nodemon ]
	npx nodemon src/server.ts
	edite script in package.json
		"start": "nodemon src/server.ts",
		npm start
	edite "main" in package.json to "dist/server.js" [for starting file]

install eslint [ helps to findout errors in app]
	npm i -D eslint
	npx eslint --init
	in .eslintrc.js edite browser to node
	"env": {
        "browser": true,
        "es2021": true
    	},
	npx eslint . --ext .ts
	script for eslint:
		"lint" : : "eslint . --ext .ts"
		npm run lint
